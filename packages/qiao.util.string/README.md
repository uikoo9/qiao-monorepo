# api
## firstLetterUpper
```javascript
'use strict';

var qiaoUtilString = require('qiao.util.string');

var test = function(){
	var str = 'table';
	var s	= qiaoUtilString.firstLetterUpper(str);
	
	console.log(s);
};

test();
```
