# api
## clear
```javascript
'use strict';

var qiao = require('qiao.util.console');

qiao.clear();
```

## writeLine
```javascript
'use strict';

var qiao = require('qiao.util.console');

qiao.clear();
qiao.writeLine(1, 'hello');
```
